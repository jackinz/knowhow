-- search text in tablename and columnname
SELECT      COLUMN_NAME AS 'ColumnName'
            ,TABLE_NAME AS  'TableName'
FROM        INFORMATION_SCHEMA.COLUMNS
WHERE       COLUMN_NAME LIKE '%MyName%'
	    or TABLE_NAME LIKE '%MyName%'
ORDER BY    TableName
            ,ColumnName;
